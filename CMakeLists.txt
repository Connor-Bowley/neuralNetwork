cmake_minimum_required(VERSION 2.8)
project( ConvNetTest )
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} "${CMAKE_SOURCE_DIR}/cmake/Modules/")
find_package( OpenCV REQUIRED )
find_library( OPENCL_LIBRARY OpenCL.framework )
SET( SRCS ConvNetTest.cpp ConvNet.cpp )
SET( CONV_LIBS "")
SET( CMAKE_CXX_FLAGS "-std=c++11")
include_directories( ${OpenCV_INCLUDE_DIRS} ${OpenCL_INCLUDE_DIRS})
add_executable( ConvNetTest ${SRCS})
add_executable( kernelBuilder kernelBuilder.cpp )
add_executable( ImageSplitter ImageSplitter.cpp )
add_executable( OpenCLTest OpenCLTest.cpp )
target_link_libraries( ConvNetTest ${OpenCV_LIBS} ${CONV_LIBS} ${OPENCL_LIBRARY})
target_link_libraries( ImageSplitter ${OpenCV_LIBS} )
target_link_libraries( OpenCLTest ${OPENCL_LIBRARY} )
target_link_libraries( kernelBuilder ${OPENCL_LIBRARY} )
